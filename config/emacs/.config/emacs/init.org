    (setq gc-cons-threshold 10000000)

    (setq default-frame-alist '((fullscreen . maximized)
                                (vertical-scroll-bars . nil)
                                (horizontal-scroll-bars . nil)
                                (ns-transparent-titlebar . t)))

    (when (and (fboundp 'startup-redirect-eln-cache)
               (fboundp 'native-comp-available-p)
               (native-comp-available-p))
      (startup-redirect-eln-cache
       (convert-standard-filename
        (expand-file-name  "var/eln-cache/" user-emacs-directory))))

    (setq package-enable-at-startup nil)

      (getenv "IN_NIX_SHELL")
      "Non-nil when running inside Nix environment.")

    (unless my/use-nix-packages
      (add-to-list 'package-archives

                   '("melpa" . "https://melpa.org/packages/"))

      (unless (package-installed-p 'leaf)
        (package-refresh-contents)
        (package-install 'leaf))

    (require 'notifications)

    (defun eshell/alert-me (&rest args)
      "Run a command and notify when it finishes."
      (let* ((cmd (string-join args " "))
    		 (buf (current-buffer))
    		 (proc (start-process-shell-command
    				"eshell-alert-me" buf cmd)))

    	;; store metadata on the process (this is the key fix)
    	(process-put proc 'cmd cmd)
    	(process-put proc 'start-time (current-time))

    	(set-process-sentinel
    	 proc
    	 (lambda (p _event)
    	   (when (memq (process-status p) '(exit signal))
    		 (let* ((cmd (process-get p 'cmd))
    				(start-time (process-get p 'start-time))
    				(elapsed (float-time
    						  (time-subtract (current-time)
    										 start-time))))
    		   (notifications-notify
    			:title "Eshell done"
    			:body (format "%.2fs: %s" elapsed cmd)
    			:app-name "Emacs")))))

    	proc))

* Local Config

#+begin_src emacs-lisp
    (defun my/load-local-config ()
      "Load machine-specific Emacs config if it exists."
      (let ((file (expand-file-name "~/.config/emacs/local.el")))
        (when (file-readable-p file)
          (condition-case err
              (progn
                (load-file file)
                (message "Loaded local config: %s" file))
            (error
             (message "Error loading local config %s: %s"
                      file err))))))

    (add-hook 'emacs-startup-hook #'my/load-local-config)
#+END_SRC
